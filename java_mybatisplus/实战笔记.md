## 分页查询

```java
@RestController
public class SysNewsController {

    @Autowired
    private SysNewsService newsService;

    @GetMapping("/list/{page}/{size}")
    public Map list(@PathVariable("page") Integer page, @PathVariable("size") Integer size){
        Page<SysNews> queryPage = new Page<>(page, size);
        Page<SysNews> resultPage = this.newsService.page(queryPage);
        long total = resultPage.getTotal();
        List<SysNews> list = resultPage.getRecords();
        Map<String,Object> map = new HashMap<>();
        map.put("total", total);
        map.put("list", list);
        return map;
    }

    @PostMapping("/add")
    public String add(@RequestBody SysNews news){
        news.setCreatetime(new Date());
        boolean save = this.newsService.save(news);
        if(save) return "success";
        return "error";
    }

    @GetMapping("/get/{id}")
    public SysNews get(@PathVariable("id") Integer id){
        return this.newsService.getById(id);
    }

    @PostMapping("/update")
    public String update(@RequestBody SysNews news){
        boolean updateById = this.newsService.updateById(news);
        if(updateById) return "success";
        return "error";
    }

    @DeleteMapping("/delete/{id}")
    public String delete(@PathVariable("id") Integer id){
        boolean removeById = this.newsService.removeById(id);
        if(removeById) return "success";
        return "error";
    }
}
```
